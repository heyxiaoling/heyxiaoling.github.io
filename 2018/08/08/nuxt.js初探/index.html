引入ivew

注意ssr需要设置为true，如果为false会出现报错警告

css: [
    { src: 'iview/dist/styles/iview.css'}
],
plugins: [
    { src: '~plugins/iview.js', ssr: true }
]

引入公共js

//plugins/common.js
import Vue from 'vue';
import axios from 'axios';
import iView from 'iview';
import {BASEPATH} from '@/config/config';
import http from './http';

export const Common = {};


Common.install = function (Vue, options) {
    Vue.prototype.http = http;
}

Vue.use(Common);

//http.js

import axios from 'axios';
import iView from 'iview';
import {BASEPATH} from '@/config/config';

const instance = axios.create({
    baseURL: BASEPATH,
    timeout: 30000
});

export default (option)=>{
    return new Promise((resolve, reject) => {
        option.withCredentials = true;
        iView.LoadingBar.start();
        instance(option).then((res)=>{
            iView.LoadingBar.finish();
            if(res.status !== 200) {
                alert('error','提示',res.data.desc);
                reject(res);
            }else{ 
                resolve(res);
            }
        }).catch((err)=>{
            iView.LoadingBar.error();
            console.log(err);
        })
    })

}

//nuxt.config.js

plugins: [
    //{ src: '~plugins/iview.js', ssr: true },
    { src: '~plugins/common.js'}
],

路由签权

使用exprss-session,body-parse插件,注意服务器路由跟前端路由路径不要相同，或者会覆盖前端的路由,我这里在页面首页使用中间件middleware,在store，actions里使用nuxtServerInit方法，注意这个方法只能在store的最外层

//nuxt.config.js
const bodyParser = require('body-parser');
const session = require('express-session');
serverMiddleware: [
    // body-parser middleware
    bodyParser.json(),
    // session middleware
    session({
        secret: 'super-secret-key',
        resave: false,
        saveUninitialized: false,
        cookie: { maxAge: 60000 }
    }),
    // Api middleware
    // We add /api/login & /api/logout routes
    '~/api'
]

//api>index.js

const express = require('express')

// Create express router
const router = express.Router()

// Transform req & res to have the same API as express
// So we can use res.status() & res.json()
var app = express()
router.use((req, res, next) => {
    Object.setPrototypeOf(req, app.request)
    Object.setPrototypeOf(res, app.response)
    req.res = res
    res.req = req
    next()
})

// Add POST - /api/login
router.post('/login', (req, res) => {
    if (req.body.username === 'demo' && req.body.password === 'demo') {
        req.session.authUser = { username: 'demo' }
        return res.json({ username: 'demo' })
    }
    res.status(401).json({ message: 'Bad credentials' })
})

// Add POST - /api/logout
router.post('/logout', (req, res) => {
    delete req.session.authUser
    res.json({ ok: true })
})

// Export the server middleware
module.exports = {
    path: '/api',
    handler: router
}

//首页做路由签权 page>index.vue

export default{

    middleware: 'auth',
}

//middleware>auth.js

export default function ({store, error, route, req, res, redirect}) {
    let redirectURL = '/users/login';
    if (!store.state.login.authUser) {
        redirect(redirectURL)
    }
})

//store>actions


export default {
    nuxtServerInit({ commit }, { req }) {
        if (req.session && req.session.authUser) {
            commit('SET_USER', req.session.authUser)
        }
    },
    async login({ commit }, { username, password }) {
        try {
            const { data } = await axios.post('/api/login', { username, password })
            commit('SET_USER', data)
        } catch (error) {
            if (error.response && error.response.status === 401) {
                throw new Error('Bad credentials')
            }
            throw error
        }
    },

    async logout({ commit }) {
        await axios.post('/api/logout')
        commit('SET_USER', null)
    }
};
