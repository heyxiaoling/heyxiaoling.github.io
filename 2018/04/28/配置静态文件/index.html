配置静态文件

github:url-loader

安装：

npm install url-loader file-loader --save-dev

在js中引入：

require("url-loader?limit=10000!./xx/x.jpg");
import homeBg1 from './image/homeBg1.jpg';

webpack配置：

{
    test: /\.(gif|jpg|png|woff|svg|eot|ttf)\??.*$/, 
    loader: 'url-loader?limit=8192'
}

我发现url-loader，只能一张一张的处理图片,gg？

how（知乎大神提出来被我看到，但是感觉好麻烦）:

const requireContext = require.context("./images", true, /^\.\/.*\.png$/);
const images = requireContext.keys().map(requireContext);

context
- How to url-loader multiple images in webpack?

var map = {
    "./moni.png": 1127,
    "./neishi.png": 1135,
    "./neiwai.png": 1136,
    "./qimian.png": 1137,
    "./wai.png": 1138
};
function webpackContext(req) {
    return __webpack_require__(webpackContextResolve(req));
};
function webpackContextResolve(req) {
    return map[req] || (function() { throw new Error("Cannot find module '" + req + "'.") }());
};
webpackContext.keys = function webpackContextKeys() {
    return Object.keys(map);
};
webpackContext.resolve = webpackContextResolve;
module.exports = webpackContext;
webpackContext.id = 1134;

这是webpack内部的实现，通过

requireContext.keys().map(requireContext)

返回的是一个images的数组，比如

[
    "http://192.168.1.164:9060/3eae22a5480a5979b97ce6dba16036d6.png",             "http://192.168.1.164:9060/571df4837e973a16cc2ea060f2570fce.png",             "http://192.168.1.164:9060/5849c45e33c05f1b4ef95bef477dd7a4.png",             "http://192.168.1.164:9060/3b70d432b0e5f69ec1e2771aa29684b9.png",             "http://192.168.1.164:9060/b81fb4bc26ebc1bd4d84d257c0292990.png"
]

直接用就可以了，需要注意的是生成最上那个map的时候会按照文件名排序。
