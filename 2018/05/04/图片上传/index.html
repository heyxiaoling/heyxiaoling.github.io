$('#camera-file').on('change',function(event){
    event.preventDefault();
    event.stopPropagation();

    var files = event.target.files;

    if(!files || files.length <= 0) return false;

    var _len = files.length;
    if(files.length > 6 || (picPaths.length + files.length) > 6) alert("最多上传六张图片！"); 

    if((picPaths.length + files.length) > 6) _len = 6 - picPaths.length;

    for(var i = 0; i < _len; i++){ 

        var imageType = /^image\//;                              
        if (!imageType.test(files[i].type)) {                   
            alert("请选择图片！");                    
            return false;                
        }

        var imageType = /^image\//;  

        if (!imageType.test(files[0].type)) {                   
            alert("请选择图片！");                    
            return false;                
        }

        var size=files[i].size/1024;

          if(size > 1024 * 20){
            alert('请您选择小于20M的图片');
            return false;
        }

        upload(files[i]);

    }

    return false;
})

//预览
function readerImage(file){
    var reader;              
    if (window.FileReader) {                    
        reader = new FileReader();                
    }else{                    
        alert("您的设备不支持图片预览功能，如需该功能请升级您的设备！");
        return false;                
    } 

    reader.onload = function(e) {
        var li = '<li><img src="' + e.target.result + '" /><span></span></li>';
        $('.image-box ul').prepend(li);

        e.preventDefault();

    };

    reader.onerror = function(e){

    }
    reader.readAsDataURL(file);
}

//上传图片，没有压缩
function upload(file,cb){
    var formData = new FormData();

    formData.append("imageType", 3);
    formData.append("userId", userId);
    formData.append("sessionId", sessionId);
    formData.append("sign",'');
    formData.append("file",file);

    $.ajax({  
        url: basePath + '/general/picUpdate',  
        type: "POST",  
        data : formData,  
        dataType:"json",
        processData: false,
        contentType: false,
        cache: false,  
        success : function(data) {
            if(data.code === 200){
                picPaths.push(data.data.url);
                localStorage.setItem('picPaths',JSON.stringify(picPaths));
                readerImage(file)
            }else{
                alert(JSON.stringify(data));
            }
        },
        error : function(error){
            let _error = JSON.stringify(error) 
            alert(_error);
            console.log(_error);
        }  
    }); 
}
