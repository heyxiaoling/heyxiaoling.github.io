配置webpack

切换到webpack2后配置方法换了好多，又重新弄了一遍

//安装
npm install webpack -g
npm install webpack --save-dev

配置转换css

less

//安装
npm install less  --save-dev;
npm install less-loader css-loader style-loader --save-dev;

//webpack配置

module: {
    rules: [
        {
            test: /\.less$/,
            use: [
              'style-loader',
              { loader: 'css-loader', options: { importLoaders: 1 } },
              'less-loader'
            ]
        }
    ]
}

//记得在js里面引入：

require(./xx/x.less);

sass

//安装：

npm install node-sass --save-dev
npm install sass-loader css-loader style-loader --save-dev

//node-sass： 安装不上先安装 cnpm

npm install -g cnpm --registry=https://registry.npm.taobao.org  
npm install node-sass --save-dev


//webpack配置：

rules: [
    {
        test: /\.scss$/,
        use: [{
            loader: "style-loader" // creates style nodes from JS strings
        }, {
            loader: "css-loader" // translates CSS into CommonJS
        }, {
            loader: "sass-loader" // compiles Sass to CSS
        }]
    }
]

//记得在js里面引入(注意后缀)：

require(./xx/x.scss);

例子：css

生成样式文件

注意：这样配置会在页面head里面生成一个style，样式都写在里面，如果你想把样式固定在一个文件，这样达到利用浏览器缓存的目的

//安装
npm install extract-text-webpack-plugin --save-dev
//webpack配置
rules: [
    {
        test: /\.less$/i,
        use: ExtractLESS.extract([ 'css-loader', 'less-loader' ])
    },
    {
        test: /\.scss$/i,
        use: ExtractLESS.extract([ 'css-loader', 'sass-loader' ])
    }
]
plugins: [
    new ExtractTextPlugin('common.css')
]
//这样会生成一个common.css文件

图片等文件处理

//安装
npm install file-loader url-loader --save-dev

//webpack

{
    test: /\.(png|jpg|jpeg|gif|eot|ttf|woff|woff2|svg|svgz)(\?.+)?$/,
    use: [
      {
        loader: 'url-loader',
        options: {
          limit: 100000
        }
      }
    ]
}

配置babel

//安装
npm install babel-core babel-loader --save-dev; //babel-core现版本为babel本身
npm install babel-preset-es2015 --save-dev //这个看你需求
npm install babel-preset-react babel-preset-stage-2 --save-dev
//因为用到react，如果不用可以不配置
//我用到react版本是15,所以配置了babel-preset-stage-2，这个视你的react具体情况而定

//安装
npm install babel-polyfill --save //兼容某些低版本浏览器对es5的不支持


//增加.babelrc文件

//在里面写上

{
    "presets": [
        ["es2015", {"modules": false}],
        "stage-2",
        "react",
    ],
    "plugins": [
        "react-hot-loader/babel"
        //react-hot
    ]
}


//webpack配置
{
    test: /\.(js|es6|jsx)$/, //注意你文件的后缀
    use: [
      'babel-loader',
    ],
    exclude: /node_modules/
},

//注意当你没有配置.babelrc文件时：

{
    test: /\.(js|es6|jsx)$/, //注意你文件的后缀
    use: [
      'babel-loader',
      options: { 
        "presets": [
            ["es2015", {"modules": false}],
            "stage-2",
            "react",
        ],
        "plugins": [
            "react-hot-loader/babel"
            //react-hot
        ]
      }
    ],
    exclude: /node_modules/
},

配置webpack-dev-server

//安装
npm install webpack-dev-server --save-dev
//react热加载,看情况配置
npm install react-hot-loader --save-dev
//配置webpack
//入口
entry: {
    app: [
        "webpack-dev-server/client?http://0.0.0.0:8888/",
        'webpack/hot/only-dev-server',
        path.resolve(__dirname, './src/app/index.es6')
    ],
},
output: {
    path: path.resolve(__dirname, './assets'),
    filename: '[name].[hash].js'
},

plugins: [
    new webpack.HotModuleReplacementPlugin(),
    // 开启全局的模块热替换(HMR)

    new webpack.NamedModulesPlugin(),
    // 当模块热替换(HMR)时在浏览器控制台输出对用户更友好的模块名字信息
]

//在package.json配置启动
"scripts": {
    "dev": "webpack-dev-server --colors --profile --env.dev"
},



//如果使用react热加载,请在入口文件添加，（这段代码情况而定）

// 模块热替换的 API
if (module.hot) {
  module.hot.accept();
}

//启动 

npm run dev

webpack-dev-middleware

npm install express webpack-dev-middleware --save-dev

var express = require("express");
var webpackDevMiddleware = require("webpack-dev-middleware");
var webpack = require("webpack");
var webpackConfig = require("./webpack.config");

var app = express();
var compiler = webpack(webpackConfig);

app.use(webpackDevMiddleware(compiler, {
  publicPath: "/" // 大部分情况下和 `output.publicPath`相同
}));

app.listen(3000, function () {
  console.log("Listening on port 3000!");
});

//默认情况下会使用watch mode。也可以使用 lazy mode，这使得 webpack 只在对入口点进行请求时再进行重新编译。

//设置仅在对入口 bundle.js 请求时进行编译：

app.use(webpackDevMiddleware(compiler, {
  lazy: true,
  filename: "bundle.js" // Same as `output.filename` in most cases.
}));
